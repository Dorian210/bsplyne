 11class MultiPatchBSplineConnectivity:
 12    """
 13    Contains all the methods to link multiple B-spline patches.
 14    It uses 3 representations of the data : 
 15      - a unique representation, possibly common with other meshes, containing 
 16        only unique nodes indices, 
 17      - a unpacked representation containing duplicated nodes indices, 
 18      - a separated representation containing duplicated nodes indices, 
 19        separated between patches. It is here for user friendliness.

 24        The indices of the unique representation needed to create the unpacked one.
 25    shape_by_patch : numpy.ndarray of int

 27    nb_nodes : int
 28        The total number of unpacked nodes.
 29    nb_unique_nodes : int
 30        The total number of unique nodes.
 31    nb_patchs : int
 32        The number of patches.
 33    npa : int
 34        The dimension of the parametric space of the B-splines.
 35    """

 62    def from_nodes_couples(cls, nodes_couples, shape_by_patch):
 63        """
 64        Create the connectivity from a list of couples of unpacked nodes.
 65
 66        Parameters
 67        ----------
 68        nodes_couples : numpy.ndarray of int
 69            Couples of indices of unpacked nodes that are considered the same.
 70            Its shape should be (# of couples, 2)
 71        shape_by_patch : numpy.ndarray of int
 72            The shape of the separated nodes by patch.
 73
 74        Returns
 75        -------
 76        MultiPatchBSplineConnectivity
 77            Instance of `MultiPatchBSplineConnectivity` created.
 78        """
 79        dic = {}
 80        for a, b in nodes_couples:
 81            if a in dic:
 82                dic[a].append(b)
 83            else:
 84                dic[a] = [b]
 85            if b in dic:
 86                dic[b].append(a)
 87            else:
 88                dic[b] = [a]
 89        nb_nodes = np.sum(np.prod(shape_by_patch, axis=1))
 90        unique_nodes_inds = np.arange(nb_nodes)
 91        for key, values in dic.items():
 92            for v in values:
 93                unique_nodes_inds[v] = unique_nodes_inds[key]
 94        different_unique_nodes_inds, inverse = np.unique(unique_nodes_inds, return_inverse=True)
 95        unique_nodes_inds -= np.cumsum(np.diff(np.concatenate(([-1], different_unique_nodes_inds))) - 1)[inverse]
 96        nb_unique_nodes = np.unique(unique_nodes_inds).size
 97        return cls(unique_nodes_inds, shape_by_patch, nb_unique_nodes)

100    def from_separated_ctrlPts(cls, separated_ctrlPts, eps=1e-10):
101        """
102        Create the connectivity from a list of control points given as 
103        a separated field by comparing every couple of points.
104
105        Parameters
106        ----------
107        separated_ctrlPts : list of numpy.ndarray of float
108            Control points of every patch to be compared in the separated 
109            representation. Every array is of shape : 
110            (``NPh``, nb elem for dim 1, ..., nb elem for dim ``npa``)
111        eps : float, optional
112            Maximum distance between two points to be considered the same, by default 1e-10
113
114        Returns
115        -------
116        MultiPatchBSplineConnectivity
117            Instance of `MultiPatchBSplineConnectivity` created.
118        """
119        NPh = separated_ctrlPts[0].shape[0]
120        assert np.all([ctrlPts.shape[0]==NPh for ctrlPts in separated_ctrlPts[1:]]), "Physical spaces must contain the same number of dimensions !"
121        shape_by_patch = np.array([ctrlPts.shape[1:] for ctrlPts in separated_ctrlPts], dtype='int')
122        nodes_couples = []
123        previous_pts = separated_ctrlPts[0].reshape((NPh, -1))
124        previous_inds_counter = previous_pts.shape[1]
125        previous_inds = np.arange(previous_inds_counter)
126        for ctrlPts in separated_ctrlPts[1:]:
127            # create current pts and inds
128            current_pts = ctrlPts.reshape((NPh, -1))
129            current_inds_counter = previous_inds_counter + current_pts.shape[1]
130            current_inds = np.arange(previous_inds_counter, current_inds_counter)
131            # get couples
132            dist = np.linalg.norm(previous_pts[:, :, None] - current_pts[:, None, :], axis=0)
133            previous_inds_inds, current_inds_inds = (dist<eps).nonzero()
134            nodes_couples.append(np.hstack((previous_inds[previous_inds_inds, None], current_inds[current_inds_inds, None])))
135            # add current to previous for next iteration
136            previous_pts = np.hstack((previous_pts, current_pts))
137            previous_inds_counter = current_inds_counter
138            previous_inds = np.hstack((previous_inds, current_inds))
139        if len(nodes_couples)>0:
140            nodes_couples = np.vstack(nodes_couples)
141        else:
142            nodes_couples = np.empty((0, 2), dtype='int')
143        return cls.from_nodes_couples(nodes_couples, shape_by_patch)

178    def pack(self, unpacked_field):
179        """
180        Extract the unique representation from an unpacked representation.
181
182        Parameters
183        ----------
184        unpacked_field : numpy.ndarray
185            The unpacked representation. Its shape should be :
186            (field, shape, ..., `self`.`nb_nodes`)
187
188        Returns
189        -------
190        unique_nodes : numpy.ndarray
191            The unique representation. Its shape is :
192            (field, shape, ..., `self`.`nb_unique_nodes`)
193        """
194        field_shape = unpacked_field.shape[:-1]
195        unique_field = np.zeros((*field_shape, self.nb_unique_nodes), dtype=unpacked_field.dtype)
196        unique_field[..., self.unique_nodes_inds[::-1]] = unpacked_field[..., ::-1]
197        return unique_field

199    def separate(self, unpacked_field):
200        """
201        Extract the separated representation from an unpacked representation.
202
203        Parameters
204        ----------
205        unpacked_field : numpy.ndarray
206            The unpacked representation. Its shape is :
207            (field, shape, ..., `self`.`nb_nodes`)
208
209        Returns
210        -------
211        separated_field : list of numpy.ndarray
212            The separated representation. Every array is of shape : 
213            (field, shape, ..., nb elem for dim 1, ..., nb elem for dim `npa`)
214        """
215        field_shape = unpacked_field.shape[:-1]
216        separated_field = []
217        ind = 0
218        for patch_shape in self.shape_by_patch:
219            next_ind = ind + np.prod(patch_shape)
220            separated_field.append(unpacked_field[..., ind:next_ind].reshape((*field_shape, *patch_shape)))
221            ind = next_ind
222        return separated_field

224    def agglomerate(self, separated_field):
225        """
226        Extract the unpacked representation from a separated representation.
227
228        Parameters
229        ----------
230        separated_field : list of numpy.ndarray
231            The separated representation. Every array is of shape : 
232            (field, shape, ..., nb elem for dim 1, ..., nb elem for dim `npa`)
233
234        Returns
235        -------
236        unpacked_field : numpy.ndarray
237            The unpacked representation. Its shape is :
238            (field, shape, ..., `self`.`nb_nodes`)
239        """
240        field_shape = separated_field[0].shape[:-self.npa]
241        assert np.all([f.shape[:-self.npa]==field_shape for f in separated_field]), "Every patch must have the same field shape !"
242        unpacked_field = np.concatenate([f.reshape((*field_shape, -1)) for f in separated_field], axis=-1)
243        return unpacked_field

245    def unique_field_indices(self, field_shape, representation="separated"):
246        """
247        Get the unique, unpacked or separated representation of a field's unique indices.
248
249        Parameters
250        ----------
251        field_shape : tuple of int
252            The shape of the field. For example, if it is a vector field, `field_shape` 
253            should be (3,). If it is a second order tensor field, it should be (3, 3).
254        representation : str, optional
255            The user must choose between `"unique"`, `"unpacked"`, and `"separated"`.
256            It corresponds to the type of representation to get, by default "separated"
257
258        Returns
259        -------
260        unique_field_indices : numpy.ndarray of int or list of numpy.ndarray of int
261            The unique, unpacked or separated representation of a field's unique indices.
262            If unique, its shape is (*`field_shape`, `self`.`nb_unique_nodes`).
263            If unpacked, its shape is : (*`field_shape`, `self`.`nb_nodes`).
264            If separated, every array is of shape : (*`field_shape`, nb elem for dim 1, ..., nb elem for dim `npa`).
265        """
266        nb_indices = np.prod(field_shape)*self.nb_unique_nodes
267        unique_field_indices_as_unique_field = np.arange(nb_indices, dtype='int').reshape((*field_shape, self.nb_unique_nodes))
268        if representation=="unique":
269            unique_field_indices = unique_field_indices_as_unique_field
270            return unique_field_indices
271        elif representation=="unpacked":
272            unique_field_indices = self.unpack(unique_field_indices_as_unique_field)
273            return unique_field_indices
274        elif representation=="separated":
275            unique_field_indices = self.separate(self.unpack(unique_field_indices_as_unique_field))
276            return unique_field_indices
277        else:
278            raise ValueError(f'Representation "{representation}" not recognised. Representation must either be "unique", "unpacked", or "separated" !')

286    def extract_exterior_borders(self, splines):
287        if self.npa<=1:
288            raise AssertionError("The parametric space must be at least 2D to extract borders !")
289        duplicate_unpacked_nodes_mask = self.get_duplicate_unpacked_nodes_mask()
290        duplicate_separated_nodes_mask = self.separate(duplicate_unpacked_nodes_mask)
291        separated_unique_nodes_inds = self.unique_field_indices(())
292        arr = np.arange(self.npa).tolist()
293        border_splines = []
294        border_unique_nodes_inds = []
295        border_shape_by_patch = []
296        for i in range(self.nb_patchs):
297            spline = splines[i]
298            duplicate_nodes_mask_spline = duplicate_separated_nodes_mask[i]
299            unique_nodes_inds_spline = separated_unique_nodes_inds[i]
300            shape_by_patch_spline = self.shape_by_patch[i]
301            for axis in range(self.npa):
302                bases = np.hstack((spline.bases[(axis + 1):], spline.bases[:axis]))
303                axes = arr[axis:-1] + arr[:axis]
304                border_shape_by_patch_spline = np.hstack((shape_by_patch_spline[(axis + 1):], shape_by_patch_spline[:axis]))
305                if not np.take(duplicate_nodes_mask_spline, 0, axis=axis).all():
306                    bspline_border = BSpline.from_bases(bases[::-1])
307                    border_splines.append(bspline_border)
308                    unique_nodes_inds_spline_border = np.take(unique_nodes_inds_spline, 0, axis=axis).transpose(axes[::-1]).ravel()
309                    border_unique_nodes_inds.append(unique_nodes_inds_spline_border)
310                    border_shape_by_patch_spline_border = border_shape_by_patch_spline[::-1][None]
311                    border_shape_by_patch.append(border_shape_by_patch_spline_border)
312                    # print(f"side {0} of axis {axis} of patch {i} uses nodes {unique_nodes_inds_spline_border}")
313                if not np.take(duplicate_nodes_mask_spline, -1, axis=axis).all():
314                    bspline_border = BSpline.from_bases(bases)
315                    border_splines.append(bspline_border)
316                    unique_nodes_inds_spline_border = np.take(unique_nodes_inds_spline, -1, axis=axis).transpose(axes).ravel()
317                    border_unique_nodes_inds.append(unique_nodes_inds_spline_border)
318                    border_shape_by_patch_spline_border = border_shape_by_patch_spline[None]
319                    border_shape_by_patch.append(border_shape_by_patch_spline_border)
320                    # print(f"side {-1} of axis {axis} of patch {i} uses nodes {unique_nodes_inds_spline_border}")
321        border_splines = np.array(border_splines, dtype='object')
322        border_unique_nodes_inds = np.concatenate(border_unique_nodes_inds)
323        border_shape_by_patch = np.concatenate(border_shape_by_patch)
324        border_unique_to_self_unique_connectivity, inverse = np.unique(border_unique_nodes_inds, return_inverse=True)
325        border_unique_nodes_inds -= np.cumsum(np.diff(np.concatenate(([-1], border_unique_to_self_unique_connectivity))) - 1)[inverse]
326        border_nb_unique_nodes = np.unique(border_unique_nodes_inds).size
327        border_connectivity = self.__class__(border_unique_nodes_inds, border_shape_by_patch, border_nb_unique_nodes)
328        return border_connectivity, border_splines, border_unique_to_self_unique_connectivity

410        new_splines = splines[patches_to_keep]
411        separated_unique_nodes_inds = self.unique_field_indices(())
412        new_unique_nodes_inds = np.concatenate([separated_unique_nodes_inds[patch].flat for patch in patches_to_keep])

417        new_connectivity = self.__class__(new_unique_nodes_inds, new_shape_by_patch, new_nb_unique_nodes)

486    @classmethod
487    def extract_border_pts(cls, field, axis, front_side, field_dim=1, offset=0):
488        npa = field.ndim - field_dim
489        base_face = np.hstack((np.arange(axis + 1, npa), np.arange(axis)))
490        if not front_side:
491            base_face = base_face[::-1]
492        border_field = field.transpose(axis + field_dim, *np.arange(field_dim), *(base_face + field_dim))[(-(1 + offset) if front_side else offset)]
493        return border_field

495    @classmethod
496    def transpose_and_flip(cls, field, transpose, flip, field_dim=1):
497        field = field.transpose(*np.arange(field_dim), *(transpose + field_dim))
498        for i in range(flip.size):
499            if flip[i]:
500                field = np.flip(field, axis=(i + field_dim))
501        return field

503    @classmethod

505        new_knots = []
506        for i in range(flip.size):

508                new_knots.append(sum(spans[i]) - knots[transpose[i]][::-1])
509            else:
510                new_knots.append(knots[transpose[i]])
511        return new_knots

595            start = end
596        nodes_couples = []
597        for i in range(self.nb_couples):

600            border_inds2_turned_and_fliped = self.__class__.transpose_and_flip(border_inds2, self.transpose_2_to_1[i], self.flip_2_to_1[i], field_dim=0)
601            nodes_couples.append(np.hstack((border_inds1.reshape((-1, 1)), border_inds2_turned_and_fliped.reshape((-1, 1)))))

 63    def from_nodes_couples(cls, nodes_couples, shape_by_patch):
 64        """
 65        Create the connectivity from a list of couples of unpacked nodes.
 66
 67        Parameters
 68        ----------
 69        nodes_couples : numpy.ndarray of int
 70            Couples of indices of unpacked nodes that are considered the same.
 71            Its shape should be (# of couples, 2)
 72        shape_by_patch : numpy.ndarray of int
 73            The shape of the separated nodes by patch.
 74
 75        Returns
 76        -------
 77        MultiPatchBSplineConnectivity
 78            Instance of `MultiPatchBSplineConnectivity` created.
 79        """
 80        dic = {}
 81        for a, b in nodes_couples:
 82            if a in dic:
 83                dic[a].append(b)
 84            else:
 85                dic[a] = [b]
 86            if b in dic:
 87                dic[b].append(a)
 88            else:
 89                dic[b] = [a]
 90        nb_nodes = np.sum(np.prod(shape_by_patch, axis=1))
 91        unique_nodes_inds = np.arange(nb_nodes)
 92        for key, values in dic.items():
 93            for v in values:
 94                unique_nodes_inds[v] = unique_nodes_inds[key]
 95        different_unique_nodes_inds, inverse = np.unique(unique_nodes_inds, return_inverse=True)
 96        unique_nodes_inds -= np.cumsum(np.diff(np.concatenate(([-1], different_unique_nodes_inds))) - 1)[inverse]
 97        nb_unique_nodes = np.unique(unique_nodes_inds).size
 98        return cls(unique_nodes_inds, shape_by_patch, nb_unique_nodes)

101    def from_separated_ctrlPts(cls, separated_ctrlPts, eps=1e-10):
102        """
103        Create the connectivity from a list of control points given as 
104        a separated field by comparing every couple of points.
105
106        Parameters
107        ----------
108        separated_ctrlPts : list of numpy.ndarray of float
109            Control points of every patch to be compared in the separated 
110            representation. Every array is of shape : 
111            (``NPh``, nb elem for dim 1, ..., nb elem for dim ``npa``)
112        eps : float, optional
113            Maximum distance between two points to be considered the same, by default 1e-10
114
115        Returns
116        -------
117        MultiPatchBSplineConnectivity
118            Instance of `MultiPatchBSplineConnectivity` created.
119        """
120        NPh = separated_ctrlPts[0].shape[0]
121        assert np.all([ctrlPts.shape[0]==NPh for ctrlPts in separated_ctrlPts[1:]]), "Physical spaces must contain the same number of dimensions !"
122        shape_by_patch = np.array([ctrlPts.shape[1:] for ctrlPts in separated_ctrlPts], dtype='int')
123        nodes_couples = []
124        previous_pts = separated_ctrlPts[0].reshape((NPh, -1))
125        previous_inds_counter = previous_pts.shape[1]
126        previous_inds = np.arange(previous_inds_counter)
127        for ctrlPts in separated_ctrlPts[1:]:
128            # create current pts and inds
129            current_pts = ctrlPts.reshape((NPh, -1))
130            current_inds_counter = previous_inds_counter + current_pts.shape[1]
131            current_inds = np.arange(previous_inds_counter, current_inds_counter)
132            # get couples
133            dist = np.linalg.norm(previous_pts[:, :, None] - current_pts[:, None, :], axis=0)
134            previous_inds_inds, current_inds_inds = (dist<eps).nonzero()
135            nodes_couples.append(np.hstack((previous_inds[previous_inds_inds, None], current_inds[current_inds_inds, None])))
136            # add current to previous for next iteration
137            previous_pts = np.hstack((previous_pts, current_pts))
138            previous_inds_counter = current_inds_counter
139            previous_inds = np.hstack((previous_inds, current_inds))
140        if len(nodes_couples)>0:
141            nodes_couples = np.vstack(nodes_couples)
142        else:
143            nodes_couples = np.empty((0, 2), dtype='int')
144        return cls.from_nodes_couples(nodes_couples, shape_by_patch)

246    def unique_field_indices(self, field_shape, representation="separated"):
247        """
248        Get the unique, unpacked or separated representation of a field's unique indices.
249
250        Parameters
251        ----------
252        field_shape : tuple of int
253            The shape of the field. For example, if it is a vector field, `field_shape` 
254            should be (3,). If it is a second order tensor field, it should be (3, 3).
255        representation : str, optional
256            The user must choose between `"unique"`, `"unpacked"`, and `"separated"`.
257            It corresponds to the type of representation to get, by default "separated"
258
259        Returns
260        -------
261        unique_field_indices : numpy.ndarray of int or list of numpy.ndarray of int
262            The unique, unpacked or separated representation of a field's unique indices.
263            If unique, its shape is (*`field_shape`, `self`.`nb_unique_nodes`).
264            If unpacked, its shape is : (*`field_shape`, `self`.`nb_nodes`).
265            If separated, every array is of shape : (*`field_shape`, nb elem for dim 1, ..., nb elem for dim `npa`).
266        """
267        nb_indices = np.prod(field_shape)*self.nb_unique_nodes
268        unique_field_indices_as_unique_field = np.arange(nb_indices, dtype='int').reshape((*field_shape, self.nb_unique_nodes))
269        if representation=="unique":
270            unique_field_indices = unique_field_indices_as_unique_field
271            return unique_field_indices
272        elif representation=="unpacked":
273            unique_field_indices = self.unpack(unique_field_indices_as_unique_field)
274            return unique_field_indices
275        elif representation=="separated":
276            unique_field_indices = self.separate(self.unpack(unique_field_indices_as_unique_field))
277            return unique_field_indices
278        else:
279            raise ValueError(f'Representation "{representation}" not recognised. Representation must either be "unique", "unpacked", or "separated" !')
