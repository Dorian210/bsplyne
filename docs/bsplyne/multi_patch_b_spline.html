 15# union-find algorithm for connectivity
 16@nb.njit(cache=True)
 17def find(parent, x):
 18    if parent[x]!=x:
 19        parent[x] = find(parent, parent[x])
 20    return parent[x]
 21
 22@nb.njit(cache=True)
 23def union(parent, rank, x, y):
 24    rootX = find(parent, x)
 25    rootY = find(parent, y)
 26    if rootX!=rootY:
 27        if rank[rootX]>rank[rootY]:
 28            parent[rootY] = rootX
 29        elif rank[rootX]<rank[rootY]:
 30            parent[rootX] = rootY
 31        else:
 32            parent[rootY] = rootX
 33            rank[rootX] += 1

 35@nb.njit(cache=True)
 36def get_unique_nodes_inds(nodes_couples, nb_nodes):
 37    parent = np.arange(nb_nodes)
 38    rank = np.zeros(nb_nodes, dtype=np.int32)
 39    for a, b in nodes_couples:
 40        union(parent, rank, a, b)
 41    unique_nodes_inds = np.empty(nb_nodes, dtype=np.int32)
 42    for i in range(nb_nodes):
 43        unique_nodes_inds[i] = find(parent, i)
 44    return unique_nodes_inds

 46class MultiPatchBSplineConnectivity:
 47    """
 48    Contains all the methods to link multiple B-spline patches.
 49    It uses 3 representations of the data : 
 50      - a unique representation, possibly common with other meshes, containing 
 51        only unique nodes indices, 
 52      - a unpacked representation containing duplicated nodes indices, 
 53      - a separated representation containing duplicated nodes indices, 
 54        separated between patches. It is here for user friendliness.
 55
 56    Attributes
 57    ----------
 58    unique_nodes_inds : numpy.ndarray of int
 59        The indices of the unique representation needed to create the unpacked one.
 60    shape_by_patch : numpy.ndarray of int
 61        The shape of the separated nodes by patch.
 62    nb_nodes : int
 63        The total number of unpacked nodes.
 64    nb_unique_nodes : int
 65        The total number of unique nodes.
 66    nb_patchs : int
 67        The number of patches.
 68    npa : int
 69        The dimension of the parametric space of the B-splines.
 70    """
 71    unique_nodes_inds: np.ndarray
 72    shape_by_patch: np.ndarray
 73    nb_nodes: int
 74    nb_unique_nodes: int
 75    nb_patchs: int
 76    npa: int

 78    def __init__(self, unique_nodes_inds, shape_by_patch, nb_unique_nodes):
 79        """
 80
 81        Parameters
 82        ----------
 83        unique_nodes_inds : numpy.ndarray of int
 84            The indices of the unique representation needed to create the unpacked one.
 85        shape_by_patch : numpy.ndarray of int
 86            The shape of the separated nodes by patch.
 87        nb_unique_nodes : int
 88            The total number of unique nodes.
 89        """
 90        self.unique_nodes_inds = unique_nodes_inds

 92        self.nb_nodes = np.sum(np.prod(self.shape_by_patch, axis=1))
 93        self.nb_unique_nodes = nb_unique_nodes #np.unique(self.unique_nodes_inds).size
 94        self.nb_patchs, self.npa = self.shape_by_patch.shape

 97    def from_nodes_couples(cls, nodes_couples, shape_by_patch):
 98        """
 99        Create the connectivity from a list of couples of unpacked nodes.
100
101        Parameters
102        ----------
103        nodes_couples : numpy.ndarray of int
104            Couples of indices of unpacked nodes that are considered the same.
105            Its shape should be (# of couples, 2)
106        shape_by_patch : numpy.ndarray of int
107            The shape of the separated nodes by patch.
108
109        Returns
110        -------
111        MultiPatchBSplineConnectivity
112            Instance of `MultiPatchBSplineConnectivity` created.
113        """
114        nb_nodes = np.sum(np.prod(shape_by_patch, axis=1))
115        unique_nodes_inds = get_unique_nodes_inds(nodes_couples, nb_nodes)
116        different_unique_nodes_inds, inverse = np.unique(unique_nodes_inds, return_inverse=True)
117        unique_nodes_inds -= np.cumsum(np.diff(np.concatenate(([-1], different_unique_nodes_inds))) - 1)[inverse]
118        nb_unique_nodes = np.unique(unique_nodes_inds).size
119        return cls(unique_nodes_inds, shape_by_patch, nb_unique_nodes)

122    def from_separated_ctrlPts(cls, separated_ctrlPts, eps=1e-10, return_nodes_couples: bool=False):
123        """
124        Create the connectivity from a list of control points given as 
125        a separated field by comparing every couple of points.
126
127        Parameters
128        ----------
129        separated_ctrlPts : list of numpy.ndarray of float
130            Control points of every patch to be compared in the separated 
131            representation. Every array is of shape : 
132            (``NPh``, nb elem for dim 1, ..., nb elem for dim ``npa``)
133        eps : float, optional
134            Maximum distance between two points to be considered the same, by default 1e-10
135        return_nodes_couples : bool, optional
136            If `True`, returns the `nodes_couples` created, by default False
137
138        Returns
139        -------
140        MultiPatchBSplineConnectivity
141            Instance of `MultiPatchBSplineConnectivity` created.
142        """
143        NPh = separated_ctrlPts[0].shape[0]
144        assert np.all([ctrlPts.shape[0]==NPh for ctrlPts in separated_ctrlPts[1:]]), "Physical spaces must contain the same number of dimensions !"
145        shape_by_patch = np.array([ctrlPts.shape[1:] for ctrlPts in separated_ctrlPts], dtype='int')
146        nodes_couples = []
147        previous_pts = separated_ctrlPts[0].reshape((NPh, -1))
148        previous_inds_counter = previous_pts.shape[1]
149        previous_inds = np.arange(previous_inds_counter)
150        for ctrlPts in separated_ctrlPts[1:]:
151            # create current pts and inds
152            current_pts = ctrlPts.reshape((NPh, -1))
153            current_inds_counter = previous_inds_counter + current_pts.shape[1]
154            current_inds = np.arange(previous_inds_counter, current_inds_counter)
155            # get couples
156            dist = np.linalg.norm(previous_pts[:, :, None] - current_pts[:, None, :], axis=0)
157            previous_inds_inds, current_inds_inds = (dist<eps).nonzero()
158            nodes_couples.append(np.hstack((previous_inds[previous_inds_inds, None], current_inds[current_inds_inds, None])))
159            # add current to previous for next iteration
160            previous_pts = np.hstack((previous_pts, current_pts))
161            previous_inds_counter = current_inds_counter
162            previous_inds = np.hstack((previous_inds, current_inds))
163        if len(nodes_couples)>0:
164            nodes_couples = np.vstack(nodes_couples)
165        else:
166            nodes_couples = np.empty((0, 2), dtype='int')
167        if return_nodes_couples:
168            return cls.from_nodes_couples(nodes_couples, shape_by_patch), nodes_couples
169        else:
170            return cls.from_nodes_couples(nodes_couples, shape_by_patch)

172    def unpack(self, unique_field):
173        """
174        Extract the unpacked representation from a unique representation.
175
176        Parameters
177        ----------
178        unique_field : numpy.ndarray
179            The unique representation. Its shape should be :
180            (field, shape, ..., `self`.`nb_unique_nodes`)
181
182        Returns
183        -------
184        unpacked_field : numpy.ndarray
185            The unpacked representation. Its shape is :
186            (field, shape, ..., `self`.`nb_nodes`)
187        """
188        unpacked_field = unique_field[..., self.unique_nodes_inds]
189        return unpacked_field

191    def pack(self, unpacked_field, method='mean'):
192        """
193        Extract the unique representation from an unpacked representation.
194
195        Parameters
196        ----------
197        unpacked_field : numpy.ndarray
198            The unpacked representation. Its shape should be :
199            (field, shape, ..., `self`.`nb_nodes`)
200        method: str
201            The method used to group values that could be different
202
203        Returns
204        -------
205        unique_nodes : numpy.ndarray
206            The unique representation. Its shape is :
207            (field, shape, ..., `self`.`nb_unique_nodes`)
208        """
209        field_shape = unpacked_field.shape[:-1]
210        unique_field = np.zeros((*field_shape, self.nb_unique_nodes), dtype=unpacked_field.dtype)
211        if method=='first':
212            unique_field[..., self.unique_nodes_inds[::-1]] = unpacked_field[..., ::-1]
213        elif method=='mean':
214            np.add.at(unique_field.T, self.unique_nodes_inds, unpacked_field.T)
215            counts = np.zeros(self.nb_unique_nodes, dtype='uint')
216            np.add.at(counts, self.unique_nodes_inds, 1)
217            unique_field /= counts
218        else:
219            raise NotImplementedError(f"Method {method} is not implemented ! Consider using 'first' or 'mean'.")
220        return unique_field

222    def separate(self, unpacked_field):
223        """
224        Extract the separated representation from an unpacked representation.
225
226        Parameters
227        ----------
228        unpacked_field : numpy.ndarray
229            The unpacked representation. Its shape is :
230            (field, shape, ..., `self`.`nb_nodes`)
231
232        Returns
233        -------
234        separated_field : list of numpy.ndarray
235            The separated representation. Every array is of shape : 
236            (field, shape, ..., nb elem for dim 1, ..., nb elem for dim `npa`)
237        """
238        field_shape = unpacked_field.shape[:-1]
239        separated_field = []
240        ind = 0
241        for patch_shape in self.shape_by_patch:
242            next_ind = ind + np.prod(patch_shape)
243            separated_field.append(unpacked_field[..., ind:next_ind].reshape((*field_shape, *patch_shape)))
244            ind = next_ind
245        return separated_field

247    def agglomerate(self, separated_field):
248        """
249        Extract the unpacked representation from a separated representation.
250
251        Parameters
252        ----------
253        separated_field : list of numpy.ndarray
254            The separated representation. Every array is of shape : 
255            (field, shape, ..., nb elem for dim 1, ..., nb elem for dim `npa`)
256
257        Returns
258        -------
259        unpacked_field : numpy.ndarray
260            The unpacked representation. Its shape is :
261            (field, shape, ..., `self`.`nb_nodes`)
262        """
263        field_shape = separated_field[0].shape[:-self.npa]
264        assert np.all([f.shape[:-self.npa]==field_shape for f in separated_field]), "Every patch must have the same field shape !"
265        unpacked_field = np.concatenate([f.reshape((*field_shape, -1)) for f in separated_field], axis=-1)
266        return unpacked_field

268    def unique_field_indices(self, field_shape, representation="separated"):
269        """
270        Get the unique, unpacked or separated representation of a field's unique indices.
271
272        Parameters
273        ----------
274        field_shape : tuple of int
275            The shape of the field. For example, if it is a vector field, `field_shape` 
276            should be (3,). If it is a second order tensor field, it should be (3, 3).
277        representation : str, optional
278            The user must choose between `"unique"`, `"unpacked"`, and `"separated"`.
279            It corresponds to the type of representation to get, by default "separated"
280
281        Returns
282        -------
283        unique_field_indices : numpy.ndarray of int or list of numpy.ndarray of int
284            The unique, unpacked or separated representation of a field's unique indices.
285            If unique, its shape is (*`field_shape`, `self`.`nb_unique_nodes`).
286            If unpacked, its shape is : (*`field_shape`, `self`.`nb_nodes`).
287            If separated, every array is of shape : (*`field_shape`, nb elem for dim 1, ..., nb elem for dim `npa`).
288        """
289        nb_indices = np.prod(field_shape)*self.nb_unique_nodes
290        unique_field_indices_as_unique_field = np.arange(nb_indices, dtype='int').reshape((*field_shape, self.nb_unique_nodes))
291        if representation=="unique":
292            unique_field_indices = unique_field_indices_as_unique_field
293            return unique_field_indices
294        elif representation=="unpacked":
295            unique_field_indices = self.unpack(unique_field_indices_as_unique_field)
296            return unique_field_indices
297        elif representation=="separated":
298            unique_field_indices = self.separate(self.unpack(unique_field_indices_as_unique_field))
299            return unique_field_indices
300        else:
301            raise ValueError(f'Representation "{representation}" not recognised. Representation must either be "unique", "unpacked", or "separated" !')

318    def extract_exterior_borders(self, splines):
319        """
320        Extract exterior borders from B-spline patches.
321
322        Parameters
323        ----------
324        splines : list[BSpline]
325            Array of B-spline patches to extract borders from.
326
327        Returns

334            Array mapping border unique nodes to original unique nodes.

336        Raises
337        ------
338        AssertionError
339            If isoparametric space dimension is less than 2.
340        """

346        arr = np.arange(self.npa).tolist()

354            shape_by_patch_spline = self.shape_by_patch[i]
355            for axis in range(self.npa):
356                bases = np.hstack((spline.bases[(axis + 1):], spline.bases[:axis]))
357                axes = arr[axis:-1] + arr[:axis]
358                border_shape_by_patch_spline = np.hstack((shape_by_patch_spline[(axis + 1):], shape_by_patch_spline[:axis]))

362                    unique_nodes_inds_spline_border = np.take(unique_nodes_inds_spline, 0, axis=axis).transpose(axes[::-1]).ravel()
363                    border_unique_nodes_inds.append(unique_nodes_inds_spline_border)
364                    border_shape_by_patch_spline_border = border_shape_by_patch_spline[::-1][None]

530        """
531        Create a subset of the multi-patch B-spline connectivity by keeping only selected patches.
532
533        Parameters
534        ----------
535        splines : list[BSpline]
536            Array of B-spline patches to subset.
537        patches_to_keep : numpy.array of int
538            Indices of patches to keep in the subset.
539
540        Returns
541        -------
542        new_connectivity : MultiPatchBSplineConnectivity
543            New connectivity object containing only the selected patches.
544        new_splines : list[BSpline]
545            Array of B-spline patches for the selected patches.
546        new_unique_to_self_unique_connectivity : numpy.ndarray of int
547            Array mapping new unique nodes to original unique nodes.
548        """
549        new_splines = splines[patches_to_keep]
550        separated_unique_nodes_inds = self.unique_field_indices(())
551        new_unique_nodes_inds = np.concatenate([separated_unique_nodes_inds[patch].flat for patch in patches_to_keep])
552        new_shape_by_patch = self.shape_by_patch[patches_to_keep]
553        new_unique_to_self_unique_connectivity, inverse = np.unique(new_unique_nodes_inds, return_inverse=True)
554        new_unique_nodes_inds -= np.cumsum(np.diff(np.concatenate(([-1], new_unique_to_self_unique_connectivity))) - 1)[inverse]

556        new_connectivity = self.__class__(new_unique_nodes_inds, new_shape_by_patch, new_nb_unique_nodes)

856    def get_operator_allxi1_to_allxi2(self, spans1, spans2, couple_ind):
857        ax1 = self.axes1[couple_ind]
858        ax2 = self.axes2[couple_ind]
859        front1 = self.front_sides1[couple_ind]
860        front2 = self.front_sides2[couple_ind]
861        transpose = self.transpose_2_to_1[couple_ind]
862        flip = self.flip_2_to_1[couple_ind]
863        
864        A = np.zeros((self.NPa, self.NPa), dtype='float')
865        A[ax2, ax1] = -1 if front1==front2 else 1
866        arr = np.arange(self.NPa)
867        j1 = np.hstack((arr[(ax1 + 1):], arr[:ax1]))
868        j2 = np.hstack((arr[(ax2 + 1):], arr[:ax2]))
869        A[j2[transpose], j1] = [-1 if f else 1 for f in flip]
870        b = np.zeros(self.NPa, dtype='float')
871        b[ax2] = (int(front1) + int(front2))*(1 if front2 else -1)
872        b[j2[transpose]] = [1 if f else 0 for f in flip]
873        
874        alpha1, beta1 = np.array(spans1).T
875        M1, p1 = np.diag(1/(beta1 - alpha1)), -alpha1/(beta1 - alpha1)
876        alpha2, beta2 = np.array(spans2).T
877        M2, p2 = np.diag(beta2 - alpha2), alpha2
878        b = p2 + M2@b + M2@A@p1
879        A = M2@A@M1
880        
881        return A, b
